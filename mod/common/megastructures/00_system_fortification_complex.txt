				planet_is_hostile_or_crisis = no

					has_owner = yes
					owner = { is_primitive = yes }

			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"

				# NOR = {

		modifier = { factor = 3 solar_system = { has_owner = yes is_same_value = owner.capital_scope.solar_system } }
