district_nsc_freebooter_hab_mining = {
	base_buildtime = @hab_time
	is_capped_by_modifier = no
	icon = district_hab_mining
	desc = district_hab_mining_desc
	show_on_uncolonized = {
		has_owner = yes
		space_owner = {
			is_lithoid_empire = yes
		}
		uses_district_set = habitat
	}
	potential = {
		has_owner = yes
		owner = {
			is_lithoid_empire = yes
		}
		uses_district_set = habitat
	}
	allow = {
		NOT = {
			has_modifier = resort_colony
		}
	}
	conversion_ratio = 1
	convert_to = {
		district_mining
	}
	resources = {
		category = planet_districts_hab_mining
		cost = {
			minerals = @hab_cost
		}
		upkeep = {
			energy = @hab_maintenance
		}
	}
	planet_modifier = {
		planet_housing_add = 2		# Less than Vanilla, this is the intended disadvantage
	}
	inline_script = {
		script = districts/habitat_districts_triggered_mods
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = {
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = {
				is_gestalt = yes
			}
		}
		job_mining_drone_add = 2
		planet_housing_add = 1
		# mult = owner.value:hab_void_dweller_jobs
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = {
				is_hive_empire = yes
			}
		}
		job_mining_drone_add = 2
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = {
				is_gestalt = no
				is_fallen_empire_spiritualist = no
			}
		}
		job_miner_add = 2		# Less than Vanilla, this is the intended disadvantage
		# mult = owner.value:hab_void_dweller_jobs
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = {
				is_fallen_empire_spiritualist = yes
			}
		}
		job_fe_acolyte_mine_add = 2
		# mult = owner.value:hab_void_dweller_jobs
	}
	triggered_desc = {
		trigger = {
			has_owner = yes
			owner = {
				is_gestalt = yes
			}
		}
		text = job_mining_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			has_owner = yes
			owner = {
				is_gestalt = no
				is_fallen_empire_spiritualist = no
			}
		}
		text = job_miner_effect_desc
	}
	triggered_desc = {
		trigger = {
			has_owner = yes
			owner = {
				is_fallen_empire_spiritualist = yes
			}
		}
		text = job_fe_acolyte_mine_effect_desc
	}
	triggered_desc = {
		trigger = {
			always = yes
		}
		text = num_mining_orbitals
	}
	ai_resource_production = {
		minerals = 10
		trigger = {
			always = yes
		}
	}
}
